Docker Compose


version: "3"
services:
  web: # Сервисов могут быть десятки
    image: username/repo:tag
    deploy:
      replicas: 5 # Сколько штук запускаем
      resources:
        limits:
          cpus: "0.1"
          memory: 50M
      restart_policy:
        condition: on-failure
    ports:
      - "4000:80"
    networks:
      - webnet
networks:
  webnet:
